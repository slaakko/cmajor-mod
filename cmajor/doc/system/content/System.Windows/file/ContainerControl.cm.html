200         [nodiscard]
201         public override Result<bool> PrintWindowTree(int level)
202         {
203             LogView* log = Application.GetLogView();
204             if (log != null)
205             {
206                 auto handleResult = ToHexString(cast<ulong>(Handle()));
207                 if (handleResult.Error())
208                 {
209                     return Result<bool>(ErrorId(handleResult.GetErrorId()));
210                 }
211                 else
212                 {
213                     const string& handleStr = handleResult.Value();
214                     auto parentTextResult = ParentText();
215                     if (parentTextResult.Error())
216                     {
217                         return Result<bool>(ErrorId(parentTextResult.GetErrorId()));
218                     }
219                     else
220                     {
221                         const string& parentText = parentTextResult.Value();
222                         auto result = log->WriteLine(string(' ', level) + "ContainerControl." + Text() + ".handle=" + handleStr + " " + 
223                             parentText + "[" + Rect(Point(), GetSize()).ToString() + "]");
224                         if (result.Error()) return result;
225                     }
226                 }
227             }
228             Component* child = children.FirstChild();
229             while (child != null)
230             {
231                 if (child is Control*)
232                 {
233                     Control* childControl = cast<Control*>(child);
234                     auto result = childControl->PrintWindowTree(level + 1);
235                     if (result.Error())
236                     {
237                         return Result<bool>(ErrorId(result.GetErrorId()));
238                     }
239                 }
240                 child = child->NextSibling();
241             }
242             return Result<bool>(true);
243         }
244         [nodiscard]
245         protected override Result<bool> ProcessMessage(Message& message)
246         {
247             switch (message.msg)
248             {
249                 case WM_SIZE:
250                 {
251                     Size newSize = message.LParamSize();
252                     Size oldSize = GetSize();
253                     if (newSize != oldSize)
254                     {
255                         Rect parentRect(Point(), newSize);
256                         int dx = newSize.w - oldSize.w;
257                         int dy = newSize.h - oldSize.h;
258                         auto result = MoveChildren(dx, dy);
259                         if (result.Error()) return result;
260                         uint windowState = cast<uint>(message.wparam);
261                         result = SetSizeInternal(newSize, windowState);
262                         if (result.Error()) return result;
263                         result = DockChildren();
264                         if (result.Error()) return result;
265                         result = OnSizeChanged(windowState);
266                         if (result.Error()) return result;
267                     }
268                     message.result = 0;
269                     return Result<bool>(true);
270                 }
271                 case WM_PAINT: case WM_MOVE: case WM_MOUSEMOVE: case WM_MOUSELEAVE: case WM_LBUTTONDOWN: case WM_LBUTTONUP: case WM_LBUTTONDBLCLK: case WM_RBUTTONDOWN: case WM_RBUTTONUP:
272                 case WM_CHAR: case WM_KEYDOWN: case WM_KEYUP: case WM_SYSCOMMAND: case WM_HSCROLL: case WM_VSCROLL: case WM_MOUSEWHEEL:
273                 case WM_SETFOCUS: case WM_KILLFOCUS: case WM_TIMER: case WM_COMMAND: case WM_SHOWWINDOW: case WM_CLIPBOARDUPDATE:
274                 {
275                     return base->ProcessMessage(message);
276                 }
277             }
278             return Result<bool>(false);
279         }

305         [nodiscard]
306         public Result<bool> DockChildren(Rect& parentRect)
307         {
308             List<Control*> filledControls;
309             List<ContainerControl*> containers;
310             Component* child = children.FirstChild();
311             while (child != null)
312             {
313                 if (child is Control*)
314                 {
315                     Control* childControl = cast<Control*>(child);
316                     switch (childControl->GetDock())
317                     {
318                         case Dock.none: break;
319                         case Dock.fill: filledControls.Add(childControl); break;
320                         default:
321                         {
322                             auto result = childControl->DockWindow(parentRect);
323                             if (result.Error()) return result;
324                             break;
325                         }
326                     }
327                     if (childControl is ContainerControl*)
328                     {
329                         ContainerControl* containerChild = cast<ContainerControl*>(childControl);
330                         containers.Add(containerChild);
331                     }
332                 }
333                 child = child->NextSibling();
334             }
335             for (Control* filledControl : filledControls)
336             {
337                 auto result = filledControl->DockWindow(parentRect);
338                 if (result.Error()) return result;
339             }
340             for (ContainerControl* container : containers)
341             {
342                 auto result = container->DockChildren();
343                 if (result.Error()) return result;
344             }
345             return Result<bool>(true);
346         }
347         [nodiscard]
348         internal override Result<bool> PaintAll(PaintEventArgs& args, bool skipMenuBar)
349         {
350             auto baseResult = base->PaintAll(args, skipMenuBar);
351             if (baseResult.Error())
352             {
353                 return Result<bool>(ErrorId(baseResult.GetErrorId()));
354             }
355             Control* topControl = TopControl();
356             Component* child = children.FirstChild();
357             while (child != null)
358             {
359                 if (child is Control*)
360                 {
361                     Control* childControl = cast<Control*>(child);
362                     if (childControl != topControl)
363                     {
364                         bool skip = false;
365                         if (skipMenuBar)
366                         {
367                             if (childControl is MenuBar*)
368                             {
369                                 skip = true;
370                             }
371                             else if (childControl is MenuBox*)
372                             {
373                                 MenuBox* menuBox = cast<MenuBox*>(childControl);
374                                 if (!menuBox->PaintThisMenuBox())
375                                 {
376                                     skip = true;
377                                 }
378                             }
379                         }
380                         if (!skip)
381                         {
382                             auto saveResult = args.graphics.Save();
383                             if (saveResult.Error())
384                             {
385                                 return Result<bool>(ErrorId(saveResult.GetErrorId()));
386                             }
387                             GraphicsState state = saveResult.Value();
388                             Result<Point> locResult = childControl->Location();
389                             if (locResult.Error())
390                             {
391                                 return Result<bool>(ErrorId(locResult.GetErrorId()));
392                             }
393                             Point loc = locResult.Value();
394                             auto childLocResult = childControl->Location();
395                             if (childLocResult.Error())
396                             {
397                                 return Result<bool>(ErrorId(childLocResult.GetErrorId()));
398                             }
399                             Point childLoc = childLocResult.Value();
400                             auto setClipResult = args.graphics.SetClip(Rect(childLoc, childControl->GetSize()));
401                             if (setClipResult.Error())
402                             {
403                                 return Result<bool>(ErrorId(setClipResult.GetErrorId()));
404                             }
405                             auto translateResult = args.graphics.TranslateTransform(loc.x, loc.y);
406                             if (translateResult.Error())
407                             {
408                                 return Result<bool>(ErrorId(translateResult.GetErrorId()));
409                             }
410                             auto childResult = childControl->PaintAll(args, skipMenuBar);
411                             if (childResult.Error())
412                             {
413                                 return Result<bool>(ErrorId(childResult.GetErrorId()));
414                             }
415                             auto restoreResult = args.graphics.Restore(state);
416                             if (restoreResult.Error())
417                             {
418                                 return Result<bool>(ErrorId(restoreResult.GetErrorId()));
419                             }
420                         }
421                     }
422                 }
423                 child = child->NextSibling();
424             }
425             if (topControl != null)
426             {
427                 auto saveResult = args.graphics.Save();
428                 if (saveResult.Error())
429                 {
430                     return Result<bool>(ErrorId(saveResult.GetErrorId()));
431                 }
432                 GraphicsState state = saveResult.Value();
433                 auto locResult = topControl->Location();
434                 if (locResult.Error())
435                 {
436                     return Result<bool>(ErrorId(locResult.GetErrorId()));
437                 }
438                 Point loc = locResult.Value();
439                 auto setClipResult = args.graphics.SetClip(Rect(loc, topControl->GetSize()));
440                 if (setClipResult.Error())
441                 {
442                     return Result<bool>(ErrorId(setClipResult.GetErrorId()));
443                 }
444                 auto translateResult = args.graphics.TranslateTransform(loc.x, loc.y);
445                 if (translateResult.Error())
446                 {
447                     return Result<bool>(ErrorId(translateResult.GetErrorId()));
448                 }
449                 auto topResult = topControl->PaintAll(args, skipMenuBar);
450                 if (topResult.Error())
451                 {
452                     return Result<bool>(ErrorId(topResult.GetErrorId()));
453                 }
454                 auto restoreResult = args.graphics.Restore(state);
455                 if (restoreResult.Error())
456                 {
457                     return Result<bool>(ErrorId(restoreResult.GetErrorId()));
458                 }
459             }
460             return Result<bool>(true);
461         }
