  6 using System;
  7 using System.Collections;

  9 namespace System.Lex

 11     public class SourceFile
 12     {
 13         public SourceFile(ustring&& content_, List<int>&& lineStartIndeces_) : content(content_), lineStartIndeces(lineStartIndeces_)
 14         {
 15         }
 16         public const ustring& Content() const
 17         {
 18             return content;
 19         }
 20         public const List<int>& LineStartIndeces() const
 21         {
 22             return lineStartIndeces;
 23         }
 24         [nodiscard]
 25         public Result<string> GetLine(int lineNumber) const
 26         {
 27             int start = lineStartIndeces[lineNumber];
 28             int len = -1;
 29             if (lineNumber < lineStartIndeces.Count() - 1)
 30             {
 31                 len = lineStartIndeces[lineNumber + 1] - start;
 32             }
 33             ustring line;
 34             if (len != -1)
 35             {
 36                 line = content.Substring(start, len);
 37             }
 38             else
 39             {
 40                 line = content.Substring(start);
 41             }
 42             ustring trimmedLine = TrimEnd(line);
 43             auto result = ToUtf8(trimmedLine);
 44             if (result.Error())
 45             {
 46                 return Result<string>(ErrorId(result.GetErrorId()));
 47             }
 48             return Result<string>(result.Value());
 49         }
 50         private ustring content;
 51         private List<int> lineStartIndeces;
 52     }
 53 
 54     public class FileMap
 55     {
 56         public int MapFileName(const string& fileName)
 57         {
 58             int fileIndex = cast<int>(fileNames.Count());
 59             fileNames.Add(fileName);
 60             return fileIndex;
 61         }
 62         public bool HasFileName(int fileIndex) const
 63         {
 64             return fileIndex >= 0 && fileIndex < fileNames.Count();
 65         }
 66         public const List<string>& FileNames() const
 67         {
 68             return fileNames;
 69         }
 70         public const string& GetFileName(int fileIndex) const
 71         {
 72             return fileNames[fileIndex];
 73         }
 74         public void AddSourceFile(int fileIndex, ustring&& content, List<int>&& lineStartIndeces)
 75         {
 76             sourceFileMap[fileIndex] = SourceFile(content, lineStartIndeces);
 77         }
 78         public SourceFile* GetSourceFile(int fileIndex) const
 79         {
 80             auto it = sourceFileMap.Find(fileIndex);
 81             if (it != sourceFileMap.End())
 82             {
 83                 return &(it->second);
 84             }
 85             else
 86             {
 87                 return null;
 88             }
 89         }
 90         private List<string> fileNames;
 91         private Map<int, SourceFile> sourceFileMap;
 92     }

 94     [nodiscard]
 95     public Result<string> MakeMessage(const string& message, const Span& span, int fileIndex, FileMap& fileMap)
 96     {
 97         string msg = message;
 98         string fileName;
 99         if (fileMap.HasFileName(fileIndex))
100         {
101             fileName = fileMap.GetFileName(fileIndex);
102         }
103         SourceFile* sourceFile = fileMap.GetSourceFile(fileIndex);
104         if (sourceFile != null)
105         {
106             LineColLen lineColLen = SpanToLineColLen(span, sourceFile->LineStartIndeces());
107             System.IO.StringWriter writer;
108             auto lineResult = sourceFile->GetLine(lineColLen.line);
109             if (lineResult.Error())
110             {
111                 return Result<string>(ErrorId(lineResult.GetErrorId()));
112             }
113             writer << message << ": " << fileName << ":" << lineColLen.line << ":\n" << lineResult.Value() << "\n" << 
114                 string(' ', lineColLen.col - 1) << string('^', lineColLen.len) << endl();
115             msg = writer.GetString();
116         }
117         return Result<string>(msg);
118     }
