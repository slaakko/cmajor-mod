 10     public class IOBuffer
 11     {
 12         public IOBuffer(long size_) : size(size_), mem(MemAlloc(size)) {}
 13         suppress IOBuffer(const IOBuffer&);
 14         suppress void operator=(const IOBuffer&);
 15         public IOBuffer(IOBuffer&& that) : size(that.size), mem(that.mem)
 16         {
 17             that.size = 0;
 18             that.mem = null;
 19         }
 20         public default void operator=(IOBuffer&& that);
 21         public ~IOBuffer()
 22         {
 23             if (mem != null)
 24             {
 25                 RtmMemFree(mem);
 26             }
 27         }
 28         public inline long Size() const
 29         {
 30             return size;
 31         }
 32         public inline void* Mem() const
 33         {
 34             return mem;
 35         }
 36         public inline byte operator[](long index) const
 37         {
 38             #assert(index >= 0 && index < size);
 39             return cast<byte*>(mem)[index];
 40         }
 41         public inline byte& operator[](long index)
 42         {
 43             #assert(index >= 0 && index < size);
 44             return cast<byte*>(mem)[index];
 45         }
 46         private long size;
 47         private void* mem;
 48     }

 50     public class BufferedStream : Stream
 51     {
 52         public BufferedStream() : base(), baseStream(null), buffer(), pos(buffer.Size()), bytesAvailable(0), end(0)
 53         {
 54         }
 55         public BufferedStream(Stream* baseStream_) : this(baseStream_, 4096)
 56         {
 57         }
 58         public BufferedStream(Stream* baseStream_, long bufferSize_) : base(), baseStream(baseStream_), buffer(bufferSize_), pos(buffer.Size()), bytesAvailable(0), end(0)
 59         {
 60         }
 61         public Stream* BaseStream()
 62         {
 63             return baseStream;
 64         }
 65         suppress BufferedStream(const BufferedStream&);
 66         suppress void operator=(const BufferedStream&);
 67         public override ~BufferedStream()
 68         {
 69             Flush();
 70         }
 71         [nodiscard]
 72         public override Result<int> ReadByte()
 73         {
 74             auto flushResult = Flush();
 75             if (flushResult.Error())
 76             {
 77                 SetErrorId(flushResult.GetErrorId());
 78                 return Result<int>(ErrorId(flushResult.GetErrorId()));
 79             }
 80             if (bytesAvailable == 0)
 81             {
 82                 auto fillResult = FillBuf();
 83                 if (fillResult.Error())
 84                 {
 85                     SetErrorId(fillResult.GetErrorId());
 86                     return Result<int>(ErrorId(fillResult.GetErrorId()));
 87                 }
 88                 if (bytesAvailable == 0)
 89                 {
 90                     return Result<int>(-1);
 91                 }
 92             }
 93             byte value = buffer[pos++];
 94             --bytesAvailable;
 95             return Result<int>(value);
 96         }
 97         [nodiscard]
 98         public override Result<long> Read(byte* buf, long count)
 99         {
100             if (count < 0)
101             {
102                 int errorId = RtmAllocateError("BufferedStream.Read: count less than zero");
103                 SetErrorId(errorId);
104                 return Result<long>(ErrorId(errorId));
105             }
106             auto flushResult = Flush();
107             if (flushResult.Error())
108             {
109                 SetErrorId(flushResult.GetErrorId());
110                 return Result<long>(ErrorId(flushResult.GetErrorId()));
111             }
112             if (bytesAvailable == 0)
113             {
114                 auto fillResult = FillBuf();
115                 if (fillResult.Error())
116                 {
117                     SetErrorId(fillResult.GetErrorId());
118                     return Result<long>(ErrorId(fillResult.GetErrorId()));
119                 }
120             }
121             long bytesRead = 0;
122             long n = Min(bytesAvailable, count);
123             for (long i = 0; i < n; ++i;)
124             {
125                 buf[i] = buffer[pos++];
126                 ++bytesRead;
127                 --bytesAvailable;
128             }
129             return Result<long>(bytesRead);
130         }
131         [nodiscard]
132         public override Result<bool> Write(byte x)
133         {
134             if (end >= buffer.Size())
135             {
136                 auto flushResult = Flush();
137                 if (flushResult.Error())
138                 {
139                     SetErrorId(flushResult.GetErrorId());
140                     return Result<bool>(ErrorId(flushResult.GetErrorId()));
141                 }
142             }
143             buffer[end++] = x;
144             return Result<bool>(true);
145         }
146         [nodiscard]
147         public override Result<bool> Write(byte* buf, long count)
148         {
149             if (count < 0)
150             {
151                 int errorId = RtmAllocateError("BufferedStream.Write: count less than zero");
152                 SetErrorId(errorId);
153                 return Result<bool>(ErrorId(errorId));
154             }
155             for (long i = 0; i < count; ++i;)
156             {
157                 auto result = Write(buf[i]);
158                 if (result.Error())
159                 {
160                     SetErrorId(result.GetErrorId());
161                     return result;
162                 }
163             }
164             return Result<bool>(true);
165         }
166         public override Result<bool> Flush()
167         {
168             if (end != 0)
169             {
170                 if (baseStream == null)
171                 {
172                     int errorId = RtmAllocateError("BufferedStream.Flush: base stream is null");
173                     SetErrorId(errorId);
174                     return Result<bool>(ErrorId(errorId));
175                 }
176                 auto writeResult = baseStream->Write(cast<byte*>(buffer.Mem()), end);
177                 if (writeResult.Error())
178                 {
179                     SetErrorId(writeResult.GetErrorId());
180                     return Result<bool>(ErrorId(writeResult.GetErrorId()));
181                 }
182                 auto flushResult = baseStream->Flush();
183                 if (flushResult.Error())
184                 {
185                     SetErrorId(flushResult.GetErrorId());
186                     return Result<bool>(ErrorId(flushResult.GetErrorId()));
187                 }
188                 end = 0;
189             }
190             return Result<bool>(true);
191         }
192         [nodiscard]
193         public override Result<bool> Seek(long pos, Origin origin)
194         {
195             auto flushResult = Flush();
196             if (flushResult.Error())
197             {
198                 SetErrorId(flushResult.GetErrorId());
199                 return Result<bool>(ErrorId(flushResult.GetErrorId()));
200             }
201             bytesAvailable = 0;
202             if (baseStream == null)
203             {
204                 int errorId = RtmAllocateError("BufferedStream.Seek: base stream is null");
205                 SetErrorId(errorId);
206                 return Result<bool>(ErrorId(errorId));
207             }
208             auto seekResult = baseStream->Seek(pos, origin);
209             if (seekResult.Error())
210             {
211                 SetErrorId(seekResult.GetErrorId());
212                 return seekResult;
213             }
214             return Result<bool>(true);
215         }
216         [nodiscard]
217         public override Result<long> Tell()
218         {
219             auto flushResult = Flush();
220             if (flushResult.Error())
221             {
222                 SetErrorId(flushResult.GetErrorId());
223                 return Result<long>(ErrorId(flushResult.GetErrorId()));
224             }
225             if (baseStream == null)
226             {
227                 int errorId = RtmAllocateError("BufferedStream.Tell: base stream is null");
228                 SetErrorId(errorId);
229                 return Result<long>(ErrorId(errorId));
230             }
231             Result<long> tellResult = baseStream->Tell();
232             if (tellResult.Error())
233             {
234                 SetErrorId(tellResult.GetErrorId());
235                 return tellResult;
236             }
237             return Result<long>(tellResult.Value() - bytesAvailable);
238         }
239         [nodiscard]
240         private Result<bool> FillBuf()
241         {
242             if (baseStream == null)
243             {
244                 int errorId = RtmAllocateError("BufferedStream.FillBuf: base stream is null");
245                 SetErrorId(errorId);
246                 return Result<bool>(ErrorId(errorId));
247             }
248             Result<long> readResult = baseStream->Read(cast<byte*>(buffer.Mem()), buffer.Size());
249             if (readResult.Error())
250             {
251                 SetErrorId(readResult.GetErrorId());
252                 return Result<bool>(ErrorId(readResult.GetErrorId()));
253             }
254             bytesAvailable = readResult.Value();
255             pos = 0;
256             return Result<bool>(true);
257         }
258         private Stream* baseStream;
259         private IOBuffer buffer;
260         private long pos;
261         private long bytesAvailable;
262         private long end;
263     }
