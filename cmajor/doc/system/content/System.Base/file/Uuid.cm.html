 97     public inline bool operator==(const Uuid& left, const Uuid& right)
 98     {

106     public bool operator<(const Uuid& left, const Uuid& right)
107     {
108         if (left.LeftHalf() < right.LeftHalf()) return true;
109         if (left.LeftHalf() > right.LeftHalf()) return false;
110         return left.RightHalf() < right.RightHalf();
111     }

113     [nodiscard]
114     public Result<string> ToString(const Uuid& uuid)
115     {
116         string s;
117         int index = 0;
118         for (byte x : uuid)
119         {
120             auto hexStringResult = ToHexString(x);
121             if (hexStringResult.Error())
122             {
123                 return Result<string>(ErrorId(hexStringResult.GetErrorId()));
124             }
125             auto toLowerResult = ToLower(hexStringResult.Value());
126             if (toLowerResult.Error())
127             {
128                 return Result<string>(ErrorId(toLowerResult.GetErrorId()));
129             }
130             s.Append(toLowerResult.Value());
131             if (index == 3 || index == 5 || index == 7 || index == 9)
132             {
133                 s.Append('-');
134             }
135             ++index;
136         }
137         return Result<string>(s);
138     }

140     [nodiscard]
141     public Result<Uuid> ParseUuid(const string& uuidHexString)
142     {
143         if (uuidHexString.Length() != 2 * Uuid.size + 4)
144         {
145             string errorMessage = "wrong number of hex bytes in uuid string \'" + uuidHexString + "\'." + ToString(Uuid.size) + " hex bytes + 4 hyphens expected.";
146             int errorId = RtmAllocateError(errorMessage.Chars());
147             return Result<Uuid>(ErrorId(errorId));
148         }
149         Uuid uuid;
150         int index = 0;
151         for (long i = 0; i < Uuid.size; ++i;)
152         {
153             string hexByteStr = uuidHexString.Substring(index, 2);
154             auto hexByte = ParseHexByte(hexByteStr);
155             if (hexByte.Error())
156             {
157                 return Result<Uuid>(ErrorId(hexByte.GetErrorId()));
158             }
159             uuid.data[i] = hexByte.Value();
160             ++index;
161             ++index;
162             if (i == 3 || i == 5 || i == 7 || i == 9)
163             {
164                 ++index;
165             }
166         }
167         return Result<Uuid>(uuid);
168     }
