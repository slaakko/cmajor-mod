 41         public void DestroyNonownedObjects()
 42         {
 43             List<XmlSerializableProxy*> proxies = GetProxies();
 44             List<XmlSerializableProxy*> toBeDestroyed;
 45             for (XmlSerializableProxy* proxy : proxies)
 46             {
 47                 if (!proxy->IsOwned())
 48                 {
 49                     toBeDestroyed.Add(proxy);
 50                 }
 51             }
 52             for (XmlSerializableProxy* d : toBeDestroyed)
 53             {
 54                 d->DestroyObject();
 55             }
 56             Clear();
 57         }

 77         public Result<UniquePtr<System.Xml.Document>> ToXmlDocument()
 78         {
 79             UniquePtr<System.Xml.Element> rootElement(System.Xml.MakeElement("xmlBundle"));
 80             auto result = ToString(rootObjectId);
 81             if (result.Error()) return Result<UniquePtr<System.Xml.Document>>(ErrorId(result.GetErrorId()));
 82             const string& rootObjectIdStr = result.Value();
 83             rootElement->SetAttribute("rootObjectId", rootObjectIdStr);
 84             List<XmlSerializableProxy*> proxies = GetProxies();
 85             for (XmlSerializableProxy* proxy : proxies)
 86             {
 87                 if (proxy->Object() != null)
 88                 {
 89                     auto result = proxy->ToXml("object");
 90                     if (result.Error()) return Result<UniquePtr<System.Xml.Document>>(ErrorId(result.GetErrorId()));
 91                     System.Xml.Element* element = result.Value();
 92                     UniquePtr<System.Xml.Element> serializableElement(element);
 93                     rootElement->AppendChild(serializableElement.Release());
 94                 }
 95             }
 96             UniquePtr<System.Xml.Document> document(System.Xml.MakeDocument());
 97             document->AppendChild(rootElement.Release());
 98             return document;
 99         }

104         public Result<string> ToXmlString(int indent) const
105         {
106             Result<UniquePtr<System.Xml.Document>> documentResult = ToXmlDocument();
107             if (documentResult.Error()) return Result<string>(ErrorId(documentResult.GetErrorId()));
108             System.Xml.Document* document = documentResult.Value().Get();
109             StringWriter stringWriter;
110             CodeFormatter formatter(stringWriter);
111             formatter.SetIndentSize(indent);
112             auto result = document->Write(formatter);
113             if (result.Error()) return Result<string>(ErrorId(result.GetErrorId()));
114             return Result<string>(stringWriter.GetString());
115         }
116         public void ResolveXmlPtrs()
117         {
118             List<XmlSerializableProxy*> proxies = GetProxies();
119             for (XmlSerializableProxy* proxy : proxies)
120             {
121                 List<XmlPtrBase*> ptrs = proxy->GetPtrs();
122                 for (XmlPtrBase* ptr : ptrs)
123                 {
124                     ptr->Resolve(this);
125                 }
126             }
127         }

137     public Result<UniquePtr<XmlBundle>> ToXmlBundle(const string& xmlStr, const string& systemId, System.Lex.FileMap& fileMap, bool createOwningBundle)
138     {
139         UniquePtr<XmlBundle> bundle(new XmlBundle());
140         if (createOwningBundle)
141         {
142             bundle->SetOwning();
143         }
144         Result<UniquePtr<System.Xml.Document>> documentResult = System.Xml.ParseXmlDocumentContent(xmlStr, systemId, fileMap);
145         if (documentResult.Error()) return Result<UniquePtr<XmlBundle>>(ErrorId(documentResult.GetErrorId()));
146         System.Xml.Document* document = documentResult.Value().Get();
147         Result<UniquePtr<System.XPath.NodeSet>> nodeSetResult = System.XPath.EvaluateToNodeSet("xmlBundle", document);
148         if (nodeSetResult.Error()) return Result<UniquePtr<XmlBundle>>(ErrorId(nodeSetResult.GetErrorId()));
149         System.XPath.NodeSet* nodeSet = nodeSetResult.Value().Get();
150         int count = nodeSet->Count();
151         if (count == 1)
152         {
153             System.Xml.Node* node = nodeSet->GetNode(0);
154             if (node->IsElementNode())
155             {
156                 System.Xml.Element* rootElement = cast<System.Xml.Element*>(node);
157                 string rootObjectIdAttr = rootElement->GetAttribute("rootObjectId");
158                 if (!rootObjectIdAttr.IsEmpty())
159                 {
160                     auto result = ParseUuid(rootObjectIdAttr);
161                     if (result.Error()) return Result<UniquePtr<XmlBundle>>(ErrorId(result.GetErrorId()));
162                     Uuid rootObjectId = result.Value();
163                     bundle->SetRootObjectId(rootObjectId);
164                 }
165             }
166             Result<UniquePtr<System.XPath.NodeSet>> objectNodeSetResult = System.XPath.EvaluateToNodeSet("xmlBundle/object", document);
167             if (objectNodeSetResult.Error()) return Result<UniquePtr<XmlBundle>>(ErrorId(objectNodeSetResult.GetErrorId()));
168             System.XPath.NodeSet* objectNodeSet = objectNodeSetResult.Value().Get();
169             int n = objectNodeSet->Count();
170             for (int i = 0; i < n; ++i;)
171             {
172                 System.Xml.Node* node = objectNodeSet->GetNode(i);
173                 if (node->IsElementNode())
174                 {
175                     System.Xml.Element* element = cast<System.Xml.Element*>(node);
176                     string classIdAttr = element->GetAttribute("classId");
177                     if (!classIdAttr.IsEmpty())
178                     {
179                         auto result = ParseInt(classIdAttr);
180                         if (result.Error()) return Result<UniquePtr<XmlBundle>>(ErrorId(result.GetErrorId()));
181                         int classId = result.Value();
182                         Result<XmlSerializable> xmlSerializableResult = XmlClassRegistry.Instance().CreateXmlSerializable(classId);
183                         if (xmlSerializableResult.Error()) return Result<UniquePtr<XmlBundle>>(ErrorId(xmlSerializableResult.GetErrorId()));
184                         XmlSerializable xmlSerializable = xmlSerializableResult.Value();
185                         auto fromXmlResult = xmlSerializable.FromXml(element);
186                         if (fromXmlResult.Error()) return Result<UniquePtr<XmlBundle>>(ErrorId(fromXmlResult.GetErrorId()));
187                         auto addResult = bundle->Add(xmlSerializable);
188                         if (addResult.Error()) return  Result<UniquePtr<XmlBundle>>(ErrorId(addResult.GetErrorId()));
189                     }
190                 }
191             }
192             bundle->ResolveXmlPtrs();
193         }
194         else
195         {
196             int errorId = AllocateError("single \'xmlBundle\' element expected");
197             return Result<UniquePtr<XmlBundle>>(ErrorId(errorId));
198         }
199         return Result<UniquePtr<XmlBundle>>(Rvalue(bundle));
200     }
