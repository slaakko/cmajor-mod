 6 using System;
 7 using System.IO;
 8 
 9 namespace System.Net.Sockets
10 {
11     [nodiscard]
12     public Result<bool> SendString(TcpSocket& socket, const string& s)
13     {
14         int length = cast<int>(s.Length());
15         byte[4] lengthBuffer;
16         MemoryWriter writer(&lengthBuffer[0], 4);
17         auto result = writer.Write(length);
18         if (result.Error())
19         {
20             return Result<bool>(ErrorId(result.GetErrorId()));
21         }
22         auto lengthResult = socket.SendAll(&lengthBuffer[0], 4);
23         if (lengthResult.Error())
24         {
25             return Result<bool>(ErrorId(lengthResult.GetErrorId()));
26         }
27         byte* p = cast<byte*>(cast<void*>(s.Chars()));
28         auto stringResult = socket.SendAll(p, length);
29         if (stringResult.Error())
30         {
31             return Result<bool>(ErrorId(stringResult.GetErrorId()));
32         }
33         return Result<bool>(true);
34     }
35 
36     [nodiscard]
37     public Result<string> ReceiveString(TcpSocket& socket)
38     {
39         byte[4] lengthBuffer;
40         int bytesToReceive = 4;
41         int offset = 0;
42         while (bytesToReceive > 0)
43         {
44             auto bytesReceivedResult = socket.Receive(&lengthBuffer[offset], bytesToReceive);
45             if (bytesReceivedResult.Error())
46             {
47                 return Result<string>(ErrorId(bytesReceivedResult.GetErrorId()));
48             }
49             int bytesReceived = bytesReceivedResult.Value();
50             bytesToReceive = bytesToReceive - bytesReceived;
51             offset = offset + bytesReceived;
52         }
53         MemoryReader reader(&lengthBuffer[0], 4);
54         auto lengthResult = reader.ReadInt();
55         if (lengthResult.Error())
56         {
57             return Result<string>(ErrorId(lengthResult.GetErrorId()));
58         }
59         int length = lengthResult.Value();
60         if (length == 0)
61         {
62             return Result<string>(string());
63         }
64         UniquePtr<byte> mem(cast<byte*>(MemAlloc(length)));
65         offset = 0;
66         bytesToReceive = length;
67         while (bytesToReceive > 0)
68         {
69             auto bytesReceivedResult = socket.Receive(mem.Get() + offset, bytesToReceive);
70             if (bytesReceivedResult.Error())
71             {
72                 return Result<string>(ErrorId(bytesReceivedResult.GetErrorId()));
73             }
74             int bytesReceived = bytesReceivedResult.Value();
75             bytesToReceive = bytesToReceive - bytesReceived;
76             offset = offset + bytesReceived;
77         }
78         string s(cast<const char*>(cast<void*>(mem.Get())), length);
79         return Result<string>(s);
80     }
